       PROCEDURE DIVISION.


       ENTRY-PARA.
           PERFORM MAIN-PARA
           STOP RUN.


       MAIN-PARA.
       MOVE 3 TO a_main
       MOVE 7 TO b_main
       MOVE 0 TO sum_main
       COMPUTE TEMP_0 = a_main + b_main
       PERFORM VARYING j_main FROM 0 BY 1 UNTIL NOT (j_main < TEMP_0)
       ADD sum_main TO j_main GIVING sum_main
       COMPUTE TEMP_0 = a_main + b_main
       END-PERFORM
       DIVIDE sum_main BY 2 GIVING TEMP_1 REMAINDER TEMP_1
       IF TEMP_1 = 0
       DISPLAY "Even Sum: "
       DISPLAY sum_main
       ELSE
       DISPLAY "Odd Sum: "
       DISPLAY sum_main
       END-IF
       EXIT.
