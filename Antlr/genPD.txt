       PROCEDURE DIVISION.


       ENTRY-PARA.
           PERFORM MAIN-PARA
           STOP RUN.


       check-PARA.
       COMPUTE TEMP_0 = x_check + 2 * 3
       IF TEMP_0 > 10
       DISPLAY "BIG"
       ELSE
       DISPLAY "SMALL"
       END-IF
       EXIT.


       MAIN-PARA.
       MOVE 2 TO p_main
       MOVE 4 TO q_main
       MOVE 10 TO r_main
       COMPUTE TEMP_1 = r_main + 1 - p_main + q_main
       MOVE TEMP_1 TO x_check
       PERFORM check-PARA
       EXIT.
