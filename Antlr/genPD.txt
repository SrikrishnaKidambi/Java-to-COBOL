       ENTRY-PARA.
           PERFORM MAIN-PARA

       MOVE 2 TO a_main
       MOVE 3 TO b_main
       MOVE 0 TO i_main
       MOVE 0 TO n_main

       PERFORM UNTIL NOT (i_main < TEMP_0)
       COMPUTE TEMP_1 = n_main + 2
       PERFORM VARYING j_main FROM 0 BY 1 UNTIL NOT (j_main < TEMP_1)

       ADD 1 TO i_main
       COMPUTE TEMP_1 = n_main + 2
       END-PERFORM

       END-PERFORM
